1041 template <typename S, typename T_ = T,
1042 require_assignable_t<value_type, S>* = nullptr,
1043 require_any_not_plain_type_t<T_, S>* = nullptr>
1044 inline var_value<T>& operator=(const var_value<S>& other) {
1045 arena_t<plain_type_t<T>> prev_val = vi_->val_;
1046 vi_->val_ = other.val();
1047 // no need to change any adjoints - these are just zeros before the reverse
1048 // pass
1049
1050 reverse_pass_callback(
1051 [this_vi = this->vi_, other_vi = other.vi_, prev_val]() mutable {
1052 this_vi->val_ = prev_val;
1053
1054 // we have no way of detecting aliasing between this->vi_->adj_ and
1055 // other.vi_->adj_, so we must copy adjoint before reseting to zero
1056
1057 // we can reuse prev_val instead of allocating a new matrix
1058 prev_val = this_vi->adj_;
1059 this_vi->adj_.setZero();
1060 other_vi->adj_ += prev_val;
1061 });
1062 return *this;
1063 }
