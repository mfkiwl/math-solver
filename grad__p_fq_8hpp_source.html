63template <bool calc_a, bool calc_b, bool calc_z, typename TupleT, typename Ta,
64 typename Tb, typename Tz,
65 require_all_eigen_vector_t<Ta, Tb>* = nullptr,
66 require_stan_scalar_t<Tz>* = nullptr>
67void grad_pFq_impl(TupleT&& grad_tuple, const Ta& a, const Tb& b, const Tz& z,
68 double precision, int outer_steps, int inner_steps) {
69 using std::max;
70 using scalar_t = return_type_t<Ta, Tb, Tz>;
71 using Ta_plain = plain_type_t<Ta>;
72 using Tb_plain = plain_type_t<Tb>;
73 using T_vec = Eigen::Matrix<scalar_t, -1, 1>;
74 ref_type_t<Ta> a_ref_in = a;
75 ref_type_t<Tb> b_ref_in = b;
76
77 // Replace any zero inputs with the smallest number representable, so that
78 // taking log and aggregating does not return -Inf
79 Ta_plain a_ref
80 = (a_ref_in.array() == 0).select(EPSILON, a_ref_in.array()).matrix();
81 Tb_plain b_ref
82 = (b_ref_in.array() == 0).select(EPSILON, b_ref_in.array()).matrix();
83 int a_size = a.size();
84 int b_size = b.size();
85
86 // Convergence criteria for the gradients are the same as for the
87 // Hypergeometric pFq itself
88 bool condition_1 = (a_size > (b_size + 1)) && (z != 0);
89 bool condition_2 = (a_size == (b_size + 1)) && (fabs(z) > 1);
90
91 if (condition_1 || condition_2) {
92 std::stringstream msg;
93 msg << "hypergeometric pFq gradient does not meet convergence "
94 << "conditions with given arguments. "
95 << "a: " << a_ref << ", b: " << b_ref << ", "
96 << "z: " << z;
97 throw std::domain_error(msg.str());
98 }
99
100 // As the gradients will be aggregating on the log scale, we will track the
101 // the values and the signs separately - to avoid taking the log of a
102 // negative input
103 Eigen::VectorXi a_signs = sign(value_of_rec(a_ref));
104 Eigen::VectorXi b_signs = sign(value_of_rec(b_ref));
105 int z_sign = sign(value_of_rec(z));
106
107 Ta_plain ap1 = (a.array() + 1).matrix();
108 Tb_plain bp1 = (b.array() + 1).matrix();
109 scalar_type_t<Tz> log_z = log(fabs(z));
110 scalar_type_t<Ta> a_prod = prod(a_ref);
111 scalar_type_t<Tb> b_prod = prod(b_ref);
112
113 // Only need the infinite sum for partials wrt a & b
114 if (calc_a || calc_b) {
115 double log_precision = log(precision);
116
117 T_vec da_mn = T_vec::Constant(a_size, NEGATIVE_INFTY);
118 T_vec db_mn = T_vec::Constant(b_size, NEGATIVE_INFTY);
119 T_vec da = T_vec::Constant(a_size, 0.0);
120 T_vec db = T_vec::Constant(b_size, 0.0);
121
122 int m = 0;
123 int n_iter = 2;
124
125 double lgamma_mp1 = 0;
126 double log_phammer_1m = 0;
127 double log_phammer_2m = 0;
128 T_vec log_phammer_ap1_m = T_vec::Zero(ap1.size());
129 T_vec log_phammer_bp1_m = T_vec::Zero(bp1.size());
130 Tz log_z_m = 0;
131 Ta_plain ap1m = ap1;
132 Tb_plain bp1m = bp1;
133
134 Ta_plain ap1n = ap1;
135 Tb_plain bp1n = bp1;
136 Ta_plain an = a_ref;
137 Tb_plain bn = b_ref;
138 Ta_plain ap1mn = ap1m;
139 Tb_plain bp1mn = bp1m;
140
141 double log_phammer_1n;
142 double log_phammer_2_mpn;
143 double lgamma_np1;
144 T_vec log_phammer_an(a_size);
145 T_vec log_phammer_ap1_n(a_size);
146 T_vec log_phammer_bp1_n(b_size);
147 T_vec log_phammer_bn(b_size);
148 T_vec log_phammer_ap1_mpn(a_size);
149 T_vec log_phammer_bp1_mpn(b_size);
150
151 int z_pow_m_sign = 1;
152 Eigen::VectorXi curr_signs_da(a_size);
153 Eigen::VectorXi curr_signs_db(b_size);
154 Eigen::VectorXi log_phammer_an_sign(a_size);
155 Eigen::VectorXi log_phammer_ap1n_sign(a_size);
156 Eigen::VectorXi log_phammer_bp1n_sign(b_size);
157 Eigen::VectorXi log_phammer_bn_sign(b_size);
158 Eigen::VectorXi log_phammer_ap1mpn_sign(a_size);
159 Eigen::VectorXi log_phammer_bp1mpn_sign(b_size);
160 Eigen::VectorXi log_phammer_ap1m_sign = Eigen::VectorXi::Ones(a_size);
161 Eigen::VectorXi log_phammer_bp1m_sign = Eigen::VectorXi::Ones(b_size);
162
163 // If the inner loop converges in 1 iteration, then the sum has coverged
164 // and another iteration of the outer loop is not needed
165 while ((n_iter > 1) && (m < outer_steps)) {
166 ap1n = ap1;
167 bp1n = bp1;
168 an = a_ref;
169 bn = b_ref;
170 ap1mn = ap1m;
171 bp1mn = bp1m;
172
173 int n = 0;
174 Tz log_z_mn = log_z_m;
175 int z_pow_mn_sign = z_pow_m_sign;
176 scalar_t inner_diff = 0;
177 lgamma_np1 = 0;
178
179 log_phammer_1n = 0;
180 log_phammer_an.setZero();
181 log_phammer_ap1_n.setZero();
182 log_phammer_bp1_n.setZero();
183 log_phammer_bn.setZero();
184 log_phammer_ap1_mpn = log_phammer_ap1_m;
185 log_phammer_bp1_mpn = log_phammer_bp1_m;
186 log_phammer_2_mpn = log_phammer_2m;
187 log_phammer_an_sign.setOnes();
188 log_phammer_ap1n_sign.setOnes();
189 log_phammer_bp1n_sign.setOnes();
190 log_phammer_bn_sign.setOnes();
191 log_phammer_ap1mpn_sign = log_phammer_ap1m_sign;
192 log_phammer_bp1mpn_sign = log_phammer_bp1m_sign;
193
194 while ((inner_diff > log_precision) && (n < inner_steps)) {
195 // Numerator term
196 scalar_t term1_mn = log_z_mn + sum(log_phammer_ap1_mpn) + log_phammer_1m
197 + log_phammer_1n;
198 // Denominator term
199 scalar_t term2_mn = lgamma_mp1 + lgamma_np1 + sum(log_phammer_bp1_mpn)
200 + log_phammer_2_mpn;
201 int base_sign = z_pow_mn_sign * log_phammer_ap1mpn_sign.prod()
202 * log_phammer_bp1mpn_sign.prod();
203
204 if (calc_a) {
205 // Division (on log scale) for the a & b partials
206 // Determine signs of each element
207 curr_signs_da = (base_sign * log_phammer_an_sign.array()
208 * log_phammer_ap1n_sign.array())
209 .matrix();
210 da_mn = (term1_mn + log_phammer_an.array())
211 - (term2_mn + log_phammer_ap1_n.array());
212
213 // Aggregate the sums on the natural scale, so that the sign can be
214 // applied before aggregation
215 da += exp(da_mn).cwiseProduct(curr_signs_da);
216 }
217
218 if (calc_b) {
219 curr_signs_db = (base_sign * log_phammer_bn_sign.array()
220 * log_phammer_bp1n_sign.array())
221 .matrix();
222 db_mn = (term1_mn + log_phammer_bn.array())
223 - (term2_mn + log_phammer_bp1_n.array());
224 db += exp(db_mn).cwiseProduct(curr_signs_db);
225 }
226
227 // Series convergence assessed by whether the maximum term is
228 // smaller than the specified criteria (precision)
229 inner_diff = max(da_mn.maxCoeff(), db_mn.maxCoeff());
230
231 // Increment the input arguments and rising factorials
232 log_z_mn += log_z;
233 log_phammer_1n += log1p(n);
234 log_phammer_2_mpn += log(2 + m + n);
235
236 log_phammer_ap1_n.array()
237 += log(math::fabs((ap1n.array() == 0).select(1.0, ap1n.array())));
238 log_phammer_bp1_n.array()
239 += log(math::fabs((bp1n.array() == 0).select(1.0, bp1n.array())));
240 log_phammer_an.array()
241 += log(math::fabs((an.array() == 0).select(1.0, an.array())));
242 log_phammer_bn.array()
243 += log(math::fabs((bn.array() == 0).select(1.0, bn.array())));
244 log_phammer_ap1_mpn.array()
245 += log(math::fabs((ap1mn.array() == 0).select(1.0, ap1mn.array())));
246 log_phammer_bp1_mpn.array()
247 += log(math::fabs((bp1mn.array() == 0).select(1.0, bp1mn.array())));
248
249 z_pow_mn_sign *= z_sign;
250 log_phammer_ap1n_sign.array() *= sign(value_of_rec(ap1n)).array();
251 log_phammer_bp1n_sign.array() *= sign(value_of_rec(bp1n)).array();
252 log_phammer_an_sign.array() *= sign(value_of_rec(an)).array();
253 log_phammer_bn_sign.array() *= sign(value_of_rec(bn)).array();
254 log_phammer_ap1mpn_sign.array() *= sign(value_of_rec(ap1mn)).array();
255 log_phammer_bp1mpn_sign.array() *= sign(value_of_rec(bp1mn)).array();
256
257 n += 1;
258 lgamma_np1 += log(n);
259 ap1n.array() += 1;
260 bp1n.array() += 1;
261 an.array() += 1;
262 bn.array() += 1;
263 ap1mn.array() += 1;
264 bp1mn.array() += 1;
265 }
266
267 z_pow_m_sign *= z_sign;
268
269 n_iter = n;
270
271 log_z_m += log_z;
272 log_phammer_1m += log1p(m);
273 log_phammer_2m += log(2 + m);
274 log_phammer_ap1_m += log(math::fabs(ap1m));
275 log_phammer_ap1m_sign.array() *= sign(value_of_rec(ap1m)).array();
276 log_phammer_bp1_m += log(math::fabs(bp1m));
277 log_phammer_bp1m_sign.array() *= sign(value_of_rec(bp1m)).array();
278
279 m += 1;
280
281 lgamma_mp1 += log(m);
282 ap1m.array() += 1;
283 bp1m.array() += 1;
284 ap1mn.array() += 1;
285 bp1mn.array() += 1;
286 }
287
288 if (m == outer_steps) {
289 throw_domain_error("grad_pFq", "k (internal counter)", outer_steps,
290 "exceeded ",
291 " iterations, hypergeometric function gradient "
292 "did not converge.");
293 }
294
295 if (calc_a) {
296 auto pre_mult_a = (z * a_prod / a_ref.array() / b_prod).matrix();
297 std::get<0>(grad_tuple) = std::move(pre_mult_a.cwiseProduct(da));
298 }
299
300 if (calc_b) {
301 auto pre_mult_b = ((z * a_prod) / (b.array() * b_prod)).matrix();
302 std::get<1>(grad_tuple) = std::move(-pre_mult_b.cwiseProduct(db));
303 }
304 }
305
306 if (calc_z) {
307 std::get<2>(grad_tuple)
308 = std::move((a_prod / b_prod) * hypergeometric_pFq(ap1, bp1, z));
309 }
310}

331template <typename Ta, typename Tb, typename Tz>
332auto grad_pFq(const Ta& a, const Tb& b, const Tz& z, double precision = 1e-10,
333 int outer_steps = 1e6, int inner_steps = 1e6) {
334 using partials_t = partials_return_t<Ta, Tb, Tz>;
335 std::tuple<promote_scalar_t<partials_t, plain_type_t<Ta>>,
336 promote_scalar_t<partials_t, plain_type_t<Tb>>,
337 promote_scalar_t<partials_t, plain_type_t<Tz>>>
338 ret_tuple;
339 internal::grad_pFq_impl<!is_constant<Ta>::value, !is_constant<Tb>::value,
340 !is_constant<Tz>::value>(
341 ret_tuple, value_of(a), value_of(b), value_of(z), precision, outer_steps,
342 inner_steps);
343 return ret_tuple;
344}
