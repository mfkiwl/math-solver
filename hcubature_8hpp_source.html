70inline void combination(std::vector<int>& c, const int& dim, const int& p,
71 const int& x) {
72 size_t r, k = 0;
73 for (std::size_t i = 0; i < p - 1; i++) {
74 c[i] = (i != 0) ? c[i - 1] : 0;
75 do {
76 c[i]++;
77 r = choose(dim - c[i], p - (i + 1));
78 k = k + r;
79 } while (k < x);
80 k = k - r;
81 }
82 if (p > 1) {
83 c[p - 1] = c[p - 2] + x - k;
84 } else {
85 c[0] = x;
86 }
87}

98inline void combos(const int& k, const double& lambda, const int& dim,
99 std::vector<std::vector<double>>& p) {
100 std::vector<int> c(k);
101 const auto choose_dimk = choose(dim, k);
102 for (std::size_t i = 1; i != choose_dimk + 1; i++) {
103 std::vector<double> temp(dim, 0.0);
104 combination(c, dim, k, i);
105 for (std::size_t j = 0; j != k; j++) {
106 temp[c[j] - 1] = lambda;
107 }
108 p.push_back(temp);
109 }
110}

121inline void increment(std::vector<bool>& index, const int& k,
122 const double& lambda, const std::vector<int>& c,
123 std::vector<double>& temp) {
124 if (index.size() == 0) {
125 index.push_back(false);
126 for (std::size_t j = 0; j != k; j++) {
127 temp[c[j] - 1] = lambda;
128 }
129 return;
130 }
131 int first_zero = 0;
132 while ((first_zero < index.size()) && index[first_zero]) {
133 first_zero++;
134 }
135 if (first_zero == index.size()) {
136 index.flip();
137 for (std::size_t j = 0; j != index.size(); j++) {
138 temp[c[j] - 1] *= -1;
139 }
140 index.push_back(true);
141 temp[c[index.size() - 1] - 1] = -lambda;
142 } else {
143 for (std::size_t i = 0; i != first_zero + 1; i++) {
144 if (index[i]) {
145 index[i] = 0;
146 } else {
147 index[i] = 1;
148 }
149 temp[c[i] - 1] *= -1;
150 }
151 }
152}

164inline void signcombos(const int& k, const double& lambda, const int& dim,
165 std::vector<std::vector<double>>& p) {
166 std::vector<int> c(k);
167 const auto choose_dimk = choose(dim, k);
168 for (std::size_t i = 1; i != choose_dimk + 1; i++) {
169 std::vector<double> temp(dim, 0.0);
170 combination(c, dim, k, i);
171 std::vector<bool> index;
172 index.clear();
173 for (std::size_t j = 0; j != std::pow(2, k); j++) {
174 increment(index, k, lambda, c, temp);
175 p.push_back(temp);
176 }
177 }
178}

192template <typename F, typename T_pars>
193std::tuple<double, double> gauss_kronrod(const F& integrand, const double& a,
194 const double& b, T_pars& pars) {
195 std::vector<double> c(1, 0);
196 std::vector<double> cp(1, 0);
197 std::vector<double> cm(1, 0);
198 c[0] = 0.5 * (a + b);
199 double delta = 0.5 * (b - a);
200 double f0 = integrand(c, pars);
201 double I = f0 * wd7[7];
202 double Idash = f0 * gwd7[3];
203 for (std::size_t i = 0; i != 7; i++) {
204 double deltax = delta * xd7[i];
205 cp[0] = c[0] + deltax;
206 cm[0] = c[0] - deltax;
207 double fx = integrand(cp, pars);
208 double temp = integrand(cm, pars);
209 fx += temp;
210 I += fx * wd7[i];
211 if (i % 2 == 1) {
212 Idash += fx * gwd7[i / 2];
213 }
214 }
215 double V = fabs(delta);
216 I *= V;
217 Idash *= V;
218 return std::make_tuple(I, fabs(I - Idash));
219}

230inline void make_GenzMalik(const int& dim,
231 std::vector<std::vector<std::vector<double>>>& p,
232 std::vector<double>& w, std::vector<double>& wd) {
233 double l4 = std::sqrt(9 * 1.0 / 10);
234 double l2 = std::sqrt(9 * 1.0 / 70);
235 double l3 = l4;
236 double l5 = std::sqrt(9 * 1.0 / 19);
237
238 double twopn = std::pow(2, dim);
239
240 w[0] = twopn * ((12824 - 9120 * dim + 400 * dim * dim) * 1.0 / 19683);
241 w[1] = twopn * (980.0 / 6561);
242 w[2] = twopn * ((1820 - 400 * dim) * 1.0 / 19683);
243 w[3] = twopn * (200.0 / 19683);
244 w[4] = 6859.0 / 19683;
245 wd[3] = twopn * (25.0 / 729);
246 wd[2] = twopn * ((265 - 100 * dim) * 1.0 / 1458);
247 wd[1] = twopn * (245.0 / 486);
248 wd[0] = twopn * ((729 - 950 * dim + 50 * dim * dim) * 1.0 / 729);
249
250 combos(1, l2, dim, p[0]);
251 combos(1, l3, dim, p[1]);
252 signcombos(2, l4, dim, p[2]);
253 signcombos(dim, l5, dim, p[3]);
254}

273template <typename F, typename T_pars>
274std::tuple<double, double, int> integrate_GenzMalik(
275 const F& integrand, std::vector<std::vector<std::vector<double>>>& p,
276 std::vector<double>& w, std::vector<double>& wd, const int& dim,
277 const std::vector<double>& a, const std::vector<double>& b, T_pars& pars) {
278 std::vector<double> c(dim, 0);
279 std::vector<double> deltac(dim);
280
281 for (std::size_t i = 0; i != dim; i++) {
282 c[i] = (a[i] + b[i]) / 2;
283 }
284
285 for (std::size_t i = 0; i != dim; i++) {
286 deltac[i] = fabs(b[i] - a[i]) / 2;
287 }
288 double v = 1.0;
289 for (std::size_t i = 0; i != dim; i++) {
290 v *= deltac[i];
291 }
292
293 if (v == 0.0) {
294 return std::make_tuple(0.0, 0.0, 0);
295 }
296
297 double f1 = integrand(c, pars);
298 double f2 = 0.0;
299 double f3 = 0.0;
300 double twelvef1 = 12 * f1;
301
302 double maxdivdiff = 0.0;
303 std::vector<double> divdiff(dim);
304 std::vector<double> p2(dim);
305 std::vector<double> p3(dim);
306 std::vector<double> cc(dim, 0);
307
308 for (std::size_t i = 0; i != dim; i++) {
309 for (std::size_t j = 0; j != dim; j++) {
310 p2[j] = deltac[j] * p[0][i][j];
311 }
312
313 for (std::size_t j = 0; j != dim; j++) {
314 cc[j] = c[j] + p2[j];
315 }
316 double f2i = integrand(cc, pars);
317 for (std::size_t j = 0; j != dim; j++) {
318 cc[j] = c[j] - p2[j];
319 }
320 double temp = integrand(cc, pars);
321 f2i += temp;
322
323 for (std::size_t j = 0; j != dim; j++) {
324 p3[j] = deltac[j] * p[1][i][j];
325 }
326 for (std::size_t j = 0; j != dim; j++) {
327 cc[j] = c[j] + p3[j];
328 }
329 double f3i = integrand(cc, pars);
330 for (std::size_t j = 0; j != dim; j++) {
331 cc[j] = c[j] - p3[j];
332 }
333 temp = integrand(cc, pars);
334 f3i += temp;
335 f2 += f2i;
336 f3 += f3i;
337 divdiff[i] = fabs(f3i + twelvef1 - 7 * f2i);
338 }
339 std::vector<double> p4(dim);
340 double f4 = 0.0;
341 for (std::size_t i = 0; i != p[2].size(); i++) {
342 for (std::size_t j = 0; j != dim; j++) {
343 p4[j] = deltac[j] * p[2][i][j];
344 }
345 for (std::size_t j = 0; j != dim; j++) {
346 cc[j] = c[j] + p4[j];
347 }
348 double temp = integrand(cc, pars);
349 f4 += temp;
350 }
351 double f5 = 0.0;
352 std::vector<double> p5(dim);
353 for (std::size_t i = 0; i != p[3].size(); i++) {
354 for (std::size_t j = 0; j != dim; j++) {
355 p5[j] = deltac[j] * p[3][i][j];
356 }
357
358 for (std::size_t j = 0; j != dim; j++) {
359 cc[j] = c[j] + p5[j];
360 }
361 double temp = integrand(cc, pars);
362 f5 += temp;
363 }
364
365 double I = v * (w[0] * f1 + w[1] * f2 + w[2] * f3 + w[3] * f4 + w[4] * f5);
366 double Idash = v * (wd[0] * f1 + wd[1] * f2 + wd[2] * f3 + wd[3] * f4);
367 double E = fabs(I - Idash);
368
369 int kdivide = 0;
370 double deltaf = E / (std::pow(10, dim) * v);
371 for (std::size_t i = 0; i != dim; i++) {
372 double delta = divdiff[i] - maxdivdiff;
373 if (delta > deltaf) {
374 kdivide = i;
375 maxdivdiff = divdiff[i];
376 } else if ((fabs(delta) <= deltaf) && (deltac[i] > deltac[kdivide])) {
377 kdivide = i;
378 }
379 }
380 return std::make_tuple(I, E, kdivide);
381}

441template <typename F, typename T_pars>
442double hcubature(const F& integrand, const T_pars& pars, const int& dim,
443 const std::vector<double>& a, const std::vector<double>& b,
444 int& maxEval, const double& reqAbsError,
445 const double& reqRelError) {
446 if (maxEval <= 0) {
447 maxEval = 1000000;
448 }
449
450 double result, err;
451 int kdivide = 0;
452
453 std::vector<std::vector<std::vector<double>>> p(4);
454 std::vector<double> w_five(5);
455 std::vector<double> wd_four(4);
456
457 if (dim == 1) {
458 std::tie(result, err)
459 = internal::gauss_kronrod(integrand, a[0], b[0], pars);
460 } else {
461 internal::make_GenzMalik(dim, p, w_five, wd_four);
462 std::tie(result, err, kdivide) = internal::integrate_GenzMalik(
463 integrand, p, w_five, wd_four, dim, a, b, pars);
464 }
465 int numevals
466 = (dim == 1) ? 15 : 1 + 4 * dim + 2 * dim * (dim - 1) + std::pow(2, dim);
467 int evals_per_box = numevals;
468 double error = err;
469 double val = result;
470
471 if ((error <= fmax(reqRelError * fabs(val), reqAbsError))
472 || (numevals >= maxEval)) {
473 return val;
474 }
475 std::priority_queue<internal::Box> ms;
476 internal::Box box(a, b, result, err, kdivide);
477 ms.push(box);
478
479 numevals += 2 * evals_per_box;
480 while ((numevals < maxEval)
481 && (error > max(reqRelError * fabs(val), reqAbsError))
482 && std::isfinite(val)) {
483 internal::Box box = ms.top();
484 ms.pop();
485
486 double w = (box.b[box.kdiv] - box.a[box.kdiv]) / 2;
487 std::vector<double> ma(box.a);
488
489 ma[box.kdiv] += w;
490 std::vector<double> mb(box.b);
491 mb[box.kdiv] -= w;
492
493 double result_1, result_2, err_1, err_2;
494 double kdivide_1 = math::NOT_A_NUMBER;
495 double kdivide_2 = math::NOT_A_NUMBER;
496
497 if (dim == 1) {
498 std::tie(result_1, err_1)
499 = internal::gauss_kronrod(integrand, ma[0], box.b[0], pars);
500 std::tie(result_2, err_2)
501 = internal::gauss_kronrod(integrand, box.a[0], mb[0], pars);
502 } else {
503 std::tie(result_1, err_1, kdivide_1) = internal::integrate_GenzMalik(
504 integrand, p, w_five, wd_four, dim, ma, box.b, pars);
505 std::tie(result_2, err_2, kdivide_2) = internal::integrate_GenzMalik(
506 integrand, p, w_five, wd_four, dim, box.a, mb, pars);
507 }
508 internal::Box box1(ma, box.b, result_1, err_1, kdivide_1);
509 ms.push(box1);
510 internal::Box box2(box.a, mb, result_2, err_2, kdivide_2);
511 ms.push(box2);
512 val += box1.I + box2.I - box.I;
513 error += box1.E + box2.E - box.E;
514 numevals += 2 * evals_per_box;
515 }
516 val = 0.0;
517 error = 0.0;
518
519 for (; !ms.empty(); ms.pop()) {
520 internal::Box box = ms.top();
521 val += box.I;
522 error += box.E;
523 }
524 return val;
525} // hcubature
